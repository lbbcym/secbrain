```python
class RPCRetryManager:  # ✅ NEW CLASS
    async def run_with_retry(self, coro_func, *args, max_retries=3):
        for attempt in range(max_retries):
            result = await coro_func(*args, **kwargs)
            
            # ✅ NEW: Retry on transient failures
            retryable = "timeout" in str(result.revert_reason).lower()
            if retryable and attempt < max_retries - 1:

                await asyncio.sleep(wait_time)
                continue
            return result
```

```python
class RPCRetryManager:
    """Handle transient RPC failures with exponential backoff."""
    
    async def run_with_retry(
        self,
        coro_func: Callable[..., Coroutine[Any, Any, FoundryRunResult]],
        *args: Any,
        **kwargs: Any,
    ) -> FoundryRunResult:
        for attempt in range(self.max_retries):  # 3 attempts default
            result = await coro_func(*args, **kwargs)
            if retryable and attempt < max_retries - 1:
                wait_time = base_wait ** attempt  # Exponential: 2, 4, 8 seconds
                await asyncio.sleep(wait_time)
            return result
```
